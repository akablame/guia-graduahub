📌 O que é?

Hub acadêmico que centraliza a vida universitária do aluno em um só app:

  • Gestão de Rotina — disciplinas, notas, faltas, cálculo de média e previsão de nota final
  • Certificados — upload de PDF/imagem, validação manual de horas complementares, progresso visual
  • Pesquisa TCC — busca de artigos científicos na API aberta OpenAlex
  • Radar de Eventos — lista e mapa com eventos acadêmicos

// AuthService
class AuthService {
  final FirebaseAuth _auth = FirebaseAuth.instance;
  User? get currentUser => _auth.currentUser;
  Stream<User?> get authStateChanges => _auth.authStateChanges();

  Future<UserCredential> signUp(String email, String password)
    => _auth.createUserWithEmailAndPassword(email: email, password: password);
  Future<UserCredential> signIn(String email, String password)
    => _auth.signInWithEmailAndPassword(email: email, password: password);
  Future<void> signOut() => _auth.signOut();
}

// AuthProvider (excerpt)
Future<bool> login(String email, String password) async {
  _isLoading = true; _error = null; notifyListeners();
  try {
    await _auth.signIn(email, password);
    return true;
  } on FirebaseAuthException catch (e) {
    _error = _mapError(e.code); return false;
  } finally { _isLoading = false; notifyListeners(); }
}

🛡 Regras Firestore
match /users/{userId} {
  allow read, write: if request.auth.uid == userId;
}
match /certificates/{certId} {
  allow read, update, delete: if resource.data.userId == request.auth.uid;
  allow create: if request.resource.data.userId == request.auth.uid;
}
match /grades/{gradeId} {
  allow read, update, delete: if resource.data.userId == request.auth.uid;
  allow create: if request.resource.data.userId == request.auth.uid;
}
match /events/{eventId} {
  allow read: if request.auth != null;
}
